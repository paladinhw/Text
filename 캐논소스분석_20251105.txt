int LogCollector::executeMessageHandler( int msgId, struct timeval tmval, ContextInfo& context, vector<BasicData>& params )
{
	MSGHDLRMAP_I it = m_msgIdHandlerMap.find(msgId);
	//이 if문은 특이함. 나는 it.second.size() 등으로 해서 진행할 것 같은데 end랑 비교함. 이것이 나을까?
	if (it != m_msgIdHandlerMap.end()) {
		vector<RepoAdapter*> vrepo = it->second;
		for (int ctr = 0; ctr < (int)vrepo.size(); ctr++ ){
			((RepoAdapter*)vrepo[ctr])->onMessage(msgId, tmval, context, params);
		}
	}
	else {
		LOGREPO_WARN( logger, "%s : No registered handler for msgId=%d", m_processName.c_str(), msgId );
	}
	return 0;
}
